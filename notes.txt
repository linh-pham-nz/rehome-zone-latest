function getCatFact (state = '', action) {
  switch (action.type) {
    case SHOW_CAT_FACT:
      return state

    default:
      return state
  }
}

router.get('/cat-facts', (req, res) => {
  request
  .get('https://raw.githubusercontent.com/vadimdemedes/cat-facts/master/cat-facts.json')
  .end((err, result) => {
      if (err) {
        res.status(500).send(err.message)
      } else {
        res.json(result.body)
      }
  })
})
